cat > scripts/ssot/audit_addons_40_15_3.sh <<'EOF'
set -euo pipefail

DORES_ROOT="ssot/addons/dores_por_studio"
ACTIONS_ROOT="ssot/addons/actions"
OUT="ssot/registries/ADDONS_40_15_3_INDEX_OG10.yaml"

echo "== PWD =="; pwd
test -d "$DORES_ROOT" || { echo "FAIL: missing $DORES_ROOT"; exit 1; }

# (A) detectar YAML solto no root
LEGACY="$DORES_ROOT/_legacy_root_files"
mkdir -p "$LEGACY"
shopt -s nullglob
ROOT_YAMLS=("$DORES_ROOT"/DORES_S*.yaml)
ROOT_SPILL_COUNT="${#ROOT_YAMLS[@]}"
if [ "$ROOT_SPILL_COUNT" -gt 0 ]; then
  echo "WARN: root spillover ($ROOT_SPILL_COUNT) -> movendo para $LEGACY"
  for f in "${ROOT_YAMLS[@]}"; do mv -v "$f" "$LEGACY/"; done
fi

# (B) contar 15 dores por YAML
DORES_FILES=0
DORES_FAIL=0
DORES_DORID_TOTAL=0

while IFS= read -r f; do
  DORES_FILES=$((DORES_FILES+1))
  c="$(grep -c "dor_id:" "$f" 2>/dev/null || true)"
  DORES_DORID_TOTAL=$((DORES_DORID_TOTAL+c))
  if [ "$c" -ne 15 ]; then
    echo "FAIL: dores != 15 -> $f (dor_id=$c)"
    DORES_FAIL=$((DORES_FAIL+1))
  fi
done < <(find "$DORES_ROOT"/S*/ -name "*.yaml" | sort)

# (C) contar 40 actions por YAML (se existir)
ACTIONS_FILES=0
ACTIONS_FAIL=0
ACTIONS_ACTIONID_TOTAL=0
if [ -d "$ACTIONS_ROOT" ]; then
  while IFS= read -r f; do
    ACTIONS_FILES=$((ACTIONS_FILES+1))
    c="$(grep -c "action_id:" "$f" 2>/dev/null || true)"
    ACTIONS_ACTIONID_TOTAL=$((ACTIONS_ACTIONID_TOTAL+c))
    if [ "$c" -ne 40 ]; then
      echo "WARN: actions != 40 -> $f (action_id=$c)"
      ACTIONS_FAIL=$((ACTIONS_FAIL+1))
    fi
  done < <(find "$ACTIONS_ROOT"/S*/ -name "*.yaml" 2>/dev/null | sort)
fi

# (D) gerar index canônico
cat > "$OUT" <<YAML
version: "og10.addons_40_15_3.index.v1"
name: "ADDONS_40_15_3_INDEX_OG10"
paths:
  dores_root: "$DORES_ROOT"
  actions_root: "$ACTIONS_ROOT"
rules:
  dores_per_api: 15
  actions_per_api: 40
  forbid_root_yaml_in_dores_root: true
audit_snapshot:
  root_spillover_moved: $ROOT_SPILL_COUNT
  dores:
    total_files: $DORES_FILES
    dor_id_total: $DORES_DORID_TOTAL
    fails_files_not_15: $DORES_FAIL
  actions:
    total_files: $ACTIONS_FILES
    action_id_total: $ACTIONS_ACTIONID_TOTAL
    warns_files_not_40: $ACTIONS_FAIL
gate:
  # release deve FALHAR se dores != 15
  fail_if_dores_not_15: true
YAML

echo "OK: wrote $OUT"
echo "DORES_FILES=$DORES_FILES DORES_FAIL=$DORES_FAIL ROOT_SPILL_MOVED=$ROOT_SPILL_COUNT"
echo "ACTIONS_FILES=$ACTIONS_FILES ACTIONS_WARN_NOT_40=$ACTIONS_FAIL"

# gate duro: dores tem que bater 15
test "$DORES_FAIL" -eq 0 || { echo "FAIL_GATE: dores != 15"; exit 2; }

exit 0
EOF

from pathlib import Path
import os, glob, re

DORES_ROOT="ssot/addons/dores_por_studio"
ACTIONS_ROOT="ssot/addons/actions"
OUT=Path("ssot/registries/ADDONS_40_15_3_INDEX_OG10.yaml")

# contar spillover

root_spill_count = len(spill)

dores_files = len(d_files)
dores_fail = 0
dores_dorid_total = 0

for f in d_files:
    txt = Path(f).read_text(encoding="utf-8", errors="ignore")
    c = len(re.findall(r"\bdor_id:\b", txt))
    dores_dorid_total += c
    if c != 15:
        dores_fail += 1

# contar actions (se existir)
a_files = sorted(glob.glob(os.path.join(ACTIONS_ROOT, "S*", "*.yaml")))
actions_files = len(a_files)
actions_fail = 0
actions_actionid_total = 0
for f in a_files:
    txt = Path(f).read_text(encoding="utf-8", errors="ignore")
    c = len(re.findall(r"\baction_id:\b", txt))
    actions_actionid_total += c
    if c != 40:
        actions_fail += 1

y = f"""version: "og10.addons_40_15_3.index.v1"
name: "ADDONS_40_15_3_INDEX_OG10"
paths:
  dores_root: "{DORES_ROOT}"
  actions_root: "{ACTIONS_ROOT}"
rules:
  dores_per_api: 15
  actions_per_api: 40
  forbid_root_yaml_in_dores_root: true
audit_snapshot:
  root_spillover_count: {root_spill_count}
  dores:
    total_files: {dores_files}
    dor_id_total: {dores_dorid_total}
    fails_files_not_15: {dores_fail}
  actions:
    total_files: {actions_files}
    action_id_total: {actions_actionid_total}
    warns_files_not_40: {actions_fail}
gate:
  fail_if_dores_not_15: true
"""
OUT.write_text(y, encoding="utf-8")

print("SNAP: spill", root_spill_count, "dores_files", dores_files, "dores_fail", dores_fail, "actions_files", actions_files, "actions_warn", actions_fail)

# gate duro local (igual doctor): dores_fail tem que ser 0
if dores_fail != 0:
    raise SystemExit(2)
